        return this.qty * this.price
    },

        //kindly refer this keyword inside arrow : this keyword inside is different

        return this.qty * this.qty;
    }

}

    console.log(`Name ${name}`)
    console.log(`City ${city}`)
    console.log(`State ${state}`)

}

    //log all details
